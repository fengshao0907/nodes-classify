def call_nodes
    begin
        yield
    rescue Nodes::Exception => e
        puts e.to_s
        exit e.erron
    rescue => e
        puts e.to_s, e.backtrace.join("\n")
        exit Nodes::UNKNOW_ERROR 
    end
end

def echo_write_info(r)
    puts "\0033[1;34mFailed:\0033[0m\n\t%s\n" % r[:failed].join("\n\t") unless Nodes::empty?(r[:failed])
    puts "\0033[1;34mSuccessed:\0033[0m\n\t%s\n" % r[:successed].join("\n\t") unless Nodes::empty?(r[:successed])
    puts "\0033[1;33m%s submitted,%s successed.\0033[0m" % [r[:total], r[:success]]
end

def flush_buffer
    while @output_buffer.length > 0
        host, msg = *(@output_buffer.shift)
        @outputed += 1
        printf("\0033[0;36;40m%3d\0033[0;32;40m: =============== \0033[0;33;40m%-18s \0033[0;32;40m===============\0033[0m\n", @outputed, host)
        puts msg
    end
end

def list_machine(hosts)
    half_decorate = '-'*40
    @tty.printf("\n\n%s%s%s", half_decorate, 'Begin Machine List', half_decorate)
    hosts.each { |host| @tty.printf("\n%s", host) }
    @tty.printf("\n\0033[32m%d\0033[0m hosts.", hosts.length)
    @tty.printf("\n%s%s%s\n", half_decorate, '-End Machine List-', half_decorate)
end

def parse_search_res(res, human_put)
    if res['status'] == 1
        puts res['msg']
        exit Nodes::ILLEGAL_OPTION
    end

    if @options[:human_read]
        res['data'].each { |data| human_put.call(data) }
        puts "#{res['data'].length} results."
    else
        s = ''; res['data'].each { |data| s += data['name']+' ' }
        puts s.rstrip
    end
end
